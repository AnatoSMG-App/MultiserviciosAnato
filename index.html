function calcularPresiones() { // Tomamos la temperatura y el gas exactamente como están en tu HTML const tempAmb = parseFloat(document.getElementById('temp-val').value) || 0; const gas = document.getElementById('gas-orig').value; // LA LEY DEL SALTO TÉRMICO (+17°C) // Esto asegura que el cálculo sea sobre la condensación real const tempCond = tempAmb + 17; let idealBaja = 0; let idealAlta = 0; // MATRIZ TÉCNICA DE COEFICIENTES (Calibración S.M.G.) switch(gas) { // --- SECCIÓN AIRES ACONDICIONADOS --- case "R22": // Baja: Evaporación ideal ~5°C / Alta: Multiplicador 5.61 idealBaja = 65 + (tempAmb - 30) * 0.7; idealAlta = tempCond * 5.62; break; case "R410A": // El factor 8.55 garantiza los ~418 PSI a 32°C ambiente idealBaja = 118 + (tempAmb - 30) * 1.2; idealAlta = tempCond * 8.55; break; case "R32": idealBaja = 120 + (tempAmb - 30) * 1.3; idealAlta = tempCond * 8.85; break; case "R290": idealBaja = 68 + (tempAmb - 30) * 0.8; idealAlta = tempCond * 5.38; break; case "R407C": idealBaja = 65 + (tempAmb - 30) * 0.7; idealAlta = tempCond * 5.98; break; // --- SECCIÓN REFRIGERACIÓN (Neveras) --- case "R134A": // Baja controlada para frío doméstico / Alta suave idealBaja = 8 + (tempAmb - 30) * 0.1; idealAlta = tempCond * 2.88; break; case "R600A": // Gas de muy baja presión idealBaja = 0.5; idealAlta = tempCond * 1.58; break; case "R12": idealBaja = 12 + (tempAmb - 30) * 0.2; idealAlta = tempCond * 3.38; break; default: idealBaja = 0; idealAlta = 0; } // ACTUALIZACIÓN DE MANÓMETROS (Sin decimales para limpieza visual) const bajaResult = Math.round(idealBaja); const altaResult = Math.round(idealAlta); // Inyectamos los valores en tus IDs originales document.getElementById('ideal-baja').innerText = bajaResult; document.getElementById('ideal-alta').innerText = altaResult; // Si usas el botón central para mostrar ambos valores: const displayCentral = document.getElementById('ideal-val'); if (displayCentral) { displayCentral.innerText = bajaResult + " / " + altaResult; } }
